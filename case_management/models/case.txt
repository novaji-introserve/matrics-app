class Cases(models.Model):
    _name = 'case'
    _description = 'Cases'
    _rec_name = 'title'
    _inherit = ['mail.thread']

    name = fields.Char(string="Name", compute='_compute_name', store=True, tracking=True)

    title = fields.Selection([
        ('1', 'Low'),
        ('2', 'Medium'),
        ('3', 'High')
    ], string='Severity', required=False)

    cases_description = fields.Text(string='Narration', compute='_compute_description', store=True)
    transaction_reference = fields.Char(string='Transaction Reference')

    created_at = fields.Datetime(string='Event Date', required=False)

    status_id = fields.Many2one('case.status', string='Status', required=False, default=lambda self: self._default_status()
                                 )  # Added readonly=True)
    alert_id = fields.Many2one('alert', string='Alert', required=False)
    team_id = fields.Many2one('team', string='Team', required=False)
    staff_id = fields.Many2one('staff', string='Staff', required=True)
    supervisor_one_id = fields.Many2one('staff', string='Supervisor One', required=True)
    supervisor_two_id = fields.Many2one('staff', string='Supervisor Two', required=False)
    supervisor_three_id = fields.Many2one('staff', string='Supervisor Three', required=False)
    user_id = fields.Many2one('res.users', string='Customer', required=False)
    rating_id = fields.Many2one('case.rating', string='Severity', required=False, readonly=True)
    process_category_id = fields.Many2one('exception.category', string='Process Category', required=False)
    process_id = fields.Many2one('exception.process', string='Process', required=False)
    root_category_id = fields.Many2one('exception.process.type', string='Root Category', required=False)
    root_category_process_id = fields.Many2one('exception.process.type', string='Root Category Process', required=False)

    severity_level = fields.Integer(
        string="Severity Level",
        compute="_compute_severity_level",
        store=True
    )

    #response_note = fields.Integer(string='Response Note', required=False)
    response_note = fields.Text(string='Report', required=True)
    data_source = fields.Text(string="Data Source", required=False)
    attachment_ids = fields.Many2many('ir.attachment', string='Attachments')

    staff_user_id = fields.Many2one(
        'res.users',
        string='Staff User',
        related='staff_id.user_id',
        store=True,  # Optional, but recommended for performance in many cases
        readonly=True  # Optional, depending on whether you want to display it
    )

    status_name = fields.Char(string="Status Name", compute="_compute_status_name", store=True)

    start_date = fields.Datetime(string="Start Date", compute="_compute_start_date", store=True)
    end_date = fields.Datetime(string="End Date", compute="_compute_end_date", store=True)
    cases_action = fields.Text(string='Action (Text)', required=True)
    
    status_html = fields.Html(compute='_compute_status_html', string='Status')
    is_reported = fields.Boolean(string="Reported", default=False)
    
    
    def action_close_case(self):
        for case in self:
            if not case.is_reported:
                case.is_reported = True
            else:
                case.status_id = self.env.ref('case_management.case_status_closed').id



    @api.depends('status_id')
    def _compute_status_html(self):
        for record in self:
            # Ensure status_id is valid and has a name
            if record.status_id:
                color = {
                    'overdue': 'danger',
                    'closed': 'success',
                    'open': 'primary',
                }.get(record.status_id.name, 'warning')

                # Compute the HTML content for the status
                record.status_html = f'<span class="badge bg-{color}">{record.status_id.name.capitalize()}</span>'
            else:
                record.status_html = '<span class="badge bg-warning">Unknown</span>'


    @api.depends('title')
    def _compute_name(self):
        for record in self:
            label = dict(self._fields['title'].selection).get(record.title, '')
            record.name = f"{label} Case"

    @api.onchange('staff_id', 'team_id', 'status_id', 'rating_id', 'process_category_id', 'alert_id',
                    'supervisor_one_id', 'supervisor_two_id', 'supervisor_three_id', 'root_category_process_id')
    def _onchange_safe_fields(self):
        fields_to_check = [
            'staff_id', 'team_id', 'status_id', 'rating_id', 'process_id',
            'supervisor_one_id', 'supervisor_two_id', 'supervisor_three_id', 'root_category_process_id',
            'process_category_id', 'root_category_id', 'branch_id'
        ]
        for field_name in fields_to_check:
            value = getattr(self, field_name)
            if value and not value.id:
                setattr(self, field_name, False)

    @api.depends('title')
    def _compute_description(self):
        for record in self:
            if record.title == '1':
                record.cases_description = 'This is a Low case'
            elif record.title == '2':
                record.cases_description = 'This is a Medium case'
            elif record.title == '3':
                record.cases_description = 'This is a High case'
            else:
                record.cases_description = ''

    @api.model
    def _default_status(self):
        return self.env['case.status'].search([('name', '=', 'open')], limit=1).id

    @api.onchange('title')
    def _onchange_title_set_rating(self):
        if self.title:
            ref_int = int(self.title)
            rating = self.env['case.rating'].search([('ref', '=', ref_int)], limit=1)
            self.rating_id = rating
            self.supervisor_one_id = False
            self.supervisor_two_id = False
            self.supervisor_three_id = False
        else:
            self.rating_id = False

    @api.depends('rating_id.ref')
    def _compute_severity_level(self):
        for rec in self:
            rec.severity_level = rec.rating_id.ref or 0

    @api.depends('created_at')
    def _compute_start_date(self):
        for record in self:
            record.start_date = record.created_at or fields.Datetime.now()

    @api.depends('start_date')
    def _compute_end_date(self):
        for record in self:
            if record.start_date:
                record.end_date = record.start_date + timedelta(hours=1)
            else:
                record.end_date = False

    @api.model
    def _default_status(self):
        return self.env['case.status'].search([('name', '=', 'open')], limit=1).id
    
    
    @api.model
    def cron_check_overdue_cases(self):
        """ Mark cases as overdue if they've been open for more than 48 hours """
        overdue_status = self.env['case.status'].search([('name', '=', 'overdue')], limit=1)
        if not overdue_status:
            _logger.warning("Overdue status not found!")
            return

        deadline = fields.Datetime.now() - timedelta(hours=1)
        open_cases = self.search([('status_id.name', '=', 'open'), ('created_at', '<=', deadline)])
        for case in open_cases:
            case.status_id = overdue_status

    @api.depends('status_id')
    def _compute_status_name(self):
        for rec in self:
            rec.status_name = rec.status_id.name if rec.status_id else ''
    @api.model
    def create(self, vals_list):
        res = super(Cases, self).create(vals_list)
        return res
    
    
    def action_save_and_notify(self):
        vals = self._convert_to_write({name: self[name] for name in self._cache})
        new_record = self.create([vals])  # Enclose 'vals' in a list
        return new_record.action_notification_after_save()

    def action_notification_after_save(self):
        """Displays a success notification after saving."""
        message = f"Case '{self.name}' created successfully!"
        return {
            'type': 'ir.actions.client',
            'tag': 'display_notification',
            'params': {
                'message': message,
                'type': 'success',
                'sticky': False,
            }
        }


    def action_open_close_popup(self):
        self.ensure_one()
        return {
            'name': 'Close Case',
            'type': 'ir.actions.act_window',
            'res_model': 'case',
            'res_id': self.id,
            'view_mode': 'form',
            'view_id': self.env.ref('case_management.view_case_close_form').id,
            'target': 'new',
        }
